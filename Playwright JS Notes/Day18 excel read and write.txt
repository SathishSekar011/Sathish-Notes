How to update the data to excel

const ExcelJs = require('exceljs')


async function getData() {
    let output = {row:-1, column:-1}
    const workbook = new ExcelJs.Workbook()
    await workbook.xlsx.readFile("C:/Users/user/Desktop/JavaScriptDay1/UploadFiles/excelDetailsSheet.xlsx")
    const sheet = workbook.getWorksheet('Sheet1');
    sheet.eachRow((row,rowNumber)=>{
    row.eachCell((cell, colNumber)=>{
        if(cell.value === "ms"){
            console.log(rowNumber);
            console.log(colNumber);
            output.row = rowNumber
            output.column = colNumber
        }
        
        })
    })
    const cell = sheet.getCell(output.row, output.column)
    cell.value = "mahendra singh"
    await workbook.xlsx.writeFile("C:/Users/user/Desktop/JavaScriptDay1/UploadFiles/excelDetailsSheet.xlsx")


}
