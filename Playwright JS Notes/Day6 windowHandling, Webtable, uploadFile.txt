import {test, expect} from '@playwright/test'

test.only("Test_01", async({browser})=>{
  
  const context = await browser.newContext()
  const page = await context.newPage()

  const newTab = page.locator("//button[text()='New Tab']")
  
  await page.goto("https://testautomationpractice.blogspot.com/")
  const [newPage] = await Promise.all([
    context.waitForEvent('page'),// we have 3 states pending, fullfilled, rejected
    newTab.click()
  ])
  const text = await newPage.locator("(//ul//li//a)[1]").textContent()
  expect("Home").toEqual(text)
