test.only("Test_01", async({page})=>{
  const alertBtn = page.locator("[id='alertBtn']")
  const maleRadioButton = page.locator("[id='male']")
  const femaleRadioButton = page.locator("[id='female']")

  await page.goto("https://testautomationpractice.blogspot.com/")
  await maleRadioButton.check()
  await expect(maleRadioButton).toBeChecked()
  const isTrue = await maleRadioButton.isChecked()
  await expect(true).toEqual(isTrue)

  await femaleRadioButton.click()
  await expect(femaleRadioButton).toBeChecked()
  const isboolean = await femaleRadioButton.isChecked()
  await expect(true).toEqual(isboolean)

})

const dropdown = page.locator("[id='country']")

  await page.goto("https://practice.expandtesting.com/dropdown")
// to select the value
  await dropdown.selectOption("India")
// to check india is present or not
  await expect(dropdown).toHaveValue("IN")
//  get the selected value from dropdown
  const seleted = await dropdown.inputValue()
// assert the selected values
  await expect('IN').toEqual(seleted)

await page.goto('https://testautomationpractice.blogspot.com/');
   const countryDropDown = page.locator("#country")

   await countryDropDown.selectOption({value: 'uk'});
   const selectedOption = await countryDropDown.inputValue()
   await expect(selectedOption).toEqual('uk');

   await countryDropDown.selectOption({index: 1})
   const selectedOption2 = await countryDropDown.inputValue()
   await expect(selectedOption2).toEqual('canada');

   await countryDropDown.selectOption({label: 'Japan'})
   const selectedOption3 = await countryDropDown.inputValue()
   await expect(selectedOption3).toEqual('japan');

   console.log(await countryDropDown.textContent())
   const alltext = await countryDropDown.textContent()
   await expect(alltext).toContain('India')
   console.log(await countryDropDown.locator("option").count());

   await expect(await countryDropDown.locator("option").count()).toBe(10)

   const count = await countryDropDown.locator("option").count()
   
   for(let i=1;i<=count;i++){
      console.log("//select[@id='country']//option["+i+"]");
         const text = await page.locator("//select[@id='country']//option["+i+"]").textContent()
         if(text.includes('India')){
            console.log(text);
            break;
         }
   }

const alertBtn = page.locator("button#alertBtn").first();

   page.on('dialog', async dialog=>{
      console.log(dialog.message());
      expect(dialog.message()).toContain('I am an alert box!')
      await dialog.accept()
   })
   
   await alertBtn.click()
